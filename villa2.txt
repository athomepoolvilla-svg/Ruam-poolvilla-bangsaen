function buildSlider() {
  const track = document.getElementById('sliderTrack');
  const dots  = document.getElementById('psDots');
  if (track) { track.innerHTML = ''; }
  if (dots)  { dots.innerHTML  = ''; }
  photos.forEach((src, i) => {
    if (!track) return;
    const s = document.createElement('div');
    s.className = 'ps-slide';
    s.innerHTML = `<img src="${src}" alt="" onerror="this.src='https://placehold.co/720x340/e8f4f8/aaa?text=Photo'" onclick="openFs(${i})">`;
    track.appendChild(s);
    if (dots) {
      const d = document.createElement('button');
      d.className = 'ps-dot' + (i === 0 ? ' active' : '');
      d.onclick = () => psGoTo(i);
      dots.appendChild(d);
    }
  });
  psGoTo(0);
  const multi = photos.length > 1;
  document.querySelector('.ps-btn.prev').style.display = multi ? '' : 'none';
  document.querySelector('.ps-btn.next').style.display = multi ? '' : 'none';
}

function psGoTo(idx) {
  psIdx = idx;
  document.getElementById('sliderTrack').style.transform = `translateX(-${idx * 100}%)`;
  document.querySelectorAll('.ps-dot').forEach((d, i) => d.classList.toggle('active', i === idx));

// ─── Photo Tab Bar ───
function buildTabBar() {
  const bar = document.getElementById('photoTabBar');
  if (!bar) return;

  // Count photos per category
  const counts = {};
  photos.forEach((_, i) => {
    const cat = photoCatData[String(i + 1)] || '';
    if (cat) counts[cat] = (counts[cat] || 0) + 1;
  });

  // Only show tabs that have photos assigned
  let html = `<button class="ph-tab${activeCat === 'all' ? ' active' : ''}" onclick="switchPhotoCat('all',this)">ทั้งหมด <span class="ph-count">${photos.length}</span></button>`;

  Object.entries(PHOTO_CATS).forEach(([k, v]) => {
    const n = counts[k] || 0;
    if (n === 0) return; // hide empty categories
    html += `<button class="ph-tab${activeCat === k ? ' active' : ''}" onclick="switchPhotoCat('${k}',this)">${v.icon} ${v.label} <span class="ph-count">${n}</span></button>`;
  });

  bar.innerHTML = html;
}

// ─── Photo Grid ───
function renderPhotoGrid(cat) {
  const grid = document.getElementById('photoGrid');
  if (!grid) return;

  const filtered = [];
  photos.forEach((src, i) => {
    const c = photoCatData[String(i + 1)] || '';
    if (cat === 'all' || c === cat) filtered.push({ src, idx: i });
  });

  if (!filtered.length) {
    const catInfo = PHOTO_CATS[cat];
    grid.innerHTML = `<div class="photo-empty"><div class="pe-icon">${catInfo ? catInfo.icon : '🖼️'}</div>ยังไม่มีรูปในหมวด${catInfo ? catInfo.label : 'นี้'}</div>`;
    return;
  }

  grid.innerHTML = filtered.map(item =>
    `<div class="photo-grid-item" onclick="openFs(${item.idx})">
      <img src="${item.src}" alt="รูป ${item.idx + 1}" loading="lazy"
           onerror="this.src='https://placehold.co/300x200/e8f4f8/aaa?text=Photo+${item.idx + 1}'">
    </div>`
  ).join('');
}

function daysInMonth(y, m)   { return new Date(y, m, 0).getDate(); }
function firstDayOfMonth(y, m){ return new Date(y, m - 1, 1).getDay(); }

function renderCal() {
  document.getElementById('calTitle').textContent = `${MONTH_NAMES_TH[calMonth - 1]} ${calYear}`;
  document.getElementById('calHead').innerHTML = DAY_LABELS.map(d => `<div class="cal-hcell">${d}</div>`).join('');
  const body  = document.getElementById('calBody');
  const start = firstDayOfMonth(calYear, calMonth);
  const total = daysInMonth(calYear, calMonth);
  let cells = '';
  for (let i = 0; i < start; i++) cells += `<div class="cal-day empty"></div>`;
  for (let d = 1; d <= total; d++) {
    const key    = `${calYear}-${calMonth}-${d}`;
    const b      = bookings[key] || {};
    const status = b.status || 'available';
    const price  = b.price  || '';
    const priceLabel = price ? `<span class="cal-price">฿${Number(price).toLocaleString()}</span>` : '';
    const isBlocked  = (status === 'booked' || status === 'pending' || status === 'maintenance');
    cells += `<div class="cal-day ${status} ${isBlocked ? 'blocked' : ''}" ${isBlocked ? '' : `onclick="event.stopPropagation();togglePopup(this,'${status}','${price}',${PRICE_BASE})"`}>${d}${priceLabel}</div>`;
  }
  body.innerHTML = cells;
}

document.getElementById('calPrev').onclick = () => { calMonth--; if (calMonth < 1) { calMonth = 12; calYear--; } renderCal(); };
document.getElementById('calNext').onclick = () => { calMonth++; if (calMonth > 12) { calMonth = 1; calYear++; } renderCal(); };

let popupOpenDay = null, popupScrollHandler = null, dpCurrentBase = 0, dpGuestCount = 0;

function positionPopup(el) {
  const popup = document.getElementById('dayPopup');
  const r = el.getBoundingClientRect();
  const pw = popup.offsetWidth || 130, ph = popup.offsetHeight || 60;
  let left = r.left, top = r.bottom + 8;
  if (left + pw > window.innerWidth - 8) left = window.innerWidth - pw - 8;
  if (left < 8) left = 8;
  if (top + ph > window.innerHeight - 8) top = r.top - ph - 8;
  popup.style.left = left + 'px';
  popup.style.top  = top + 'px';
}

window.togglePopup = function(el, status, price, priceBase) {
  const popup = document.getElementById('dayPopup');
  if (popupOpenDay === el) { closePopup(); return; }
  popupOpenDay  = el;
  dpGuestCount  = 0;
  dpCurrentBase = price ? Number(price) : (priceBase ? Number(priceBase) : 0);
  const s = document.getElementById('dpStatus');
  s.textContent = STATUS_LABEL[status] || status;
  s.style.color = STATUS_COLOR[status] || '#333';
  document.getElementById('dpGuestCount').textContent = '0';
  document.getElementById('dpBasePrice').innerHTML = dpCurrentBase
    ? `ราคาห้อง: <b>${dpCurrentBase.toLocaleString()}</b> บาท`
    : `<span style="color:#bbb">ไม่ระบุราคา</span>`;
  updateDpTotal();
  popup.style.animation = 'none';
  popup.style.display   = 'block';
  setTimeout(() => { positionPopup(el); popup.offsetHeight; popup.style.animation = 'popIn 0.15s ease'; }, 0);
  if (popupScrollHandler) window.removeEventListener('scroll', popupScrollHandler, true);
  popupScrollHandler = () => positionPopup(el);
  window.addEventListener('scroll', popupScrollHandler, true);
};

window.changeGuest = function(dir) {
  dpGuestCount = Math.max(0, dpGuestCount + dir);
  document.getElementById('dpGuestCount').textContent = dpGuestCount;
  updateDpTotal();
};

function updateDpTotal() {
  const total = dpCurrentBase + (dpGuestCount * 500);
  const el = document.getElementById('dpTotal');
  el.innerHTML = dpCurrentBase
    ? `${total.toLocaleString()} <span style="font-size:12px;font-weight:400;color:#888">บาท</span>`
    : `<span style="font-size:13px;color:#bbb">-</span>`;
}

window.closePopup = function() {
  document.getElementById('dayPopup').style.display = 'none';
  popupOpenDay = null;
  if (popupScrollHandler) { window.removeEventListener('scroll', popupScrollHandler, true); popupScrollHandler = null; }
};
document.addEventListener('click', function(e) {
  const popup = document.getElementById('dayPopup');
  if (popup.style.display === 'block' && !popup.contains(e.target)) closePopup();
});

function renderGalSidebar() {
  const list = document.getElementById('galCatList');

  // Count per category
  const counts = {};
  galAllPhotos.forEach(p => { if (p.cat) counts[p.cat] = (counts[p.cat] || 0) + 1; });

  let html = `<button class="gal-cat-btn${galActiveCat==='all'?' active':''}" onclick="switchGalCat('all',this)">

  </button>`;

  GAL_CAT_ORDER.forEach(k => {
    const v = PHOTO_CATS[k]; if (!v) return;
    const n = counts[k] || 0; if (n === 0) return;
    html += `<button class="gal-cat-btn${galActiveCat===k?' active':''}" onclick="switchGalCat('${k}',this)">
      <span>${v.icon} ${v.label}</span><span class="gal-cat-count">${n}</span>
    </button>`;
  });

function renderGalContent(cat) {
  const inner = document.getElementById('galContentInner');
  inner.innerHTML = '';

  if (cat === 'all') {
    // Render each category as a section
    const counts = {};
    galAllPhotos.forEach(p => { if (p.cat) counts[p.cat] = (counts[p.cat] || 0) + 1; });

    // Cover / unassigned first if any
    const unassigned = galAllPhotos.filter(p => !p.cat);

    GAL_CAT_ORDER.forEach(k => {
      const v = PHOTO_CATS[k]; if (!v) return;
      const items = galAllPhotos.filter(p => p.cat === k);
      if (!items.length) return;
      inner.innerHTML += buildGalSection(v.icon + ' ' + v.label, items);
    });

    if (unassigned.length) {
      inner.innerHTML += buildGalSection('📷 ทั้งหมด', unassigned);
    }

    // If nothing categorised, just show all
    if (!inner.innerHTML) {
      inner.innerHTML += buildGalSection('📷 รูปภาพทั้งหมด', galAllPhotos);
    }

  } else {
    const v = PHOTO_CATS[cat];
    const items = galAllPhotos.filter(p => p.cat === cat);
    inner.innerHTML = buildGalSection(v ? v.icon + ' ' + v.label : cat, items);
  }
}

function buildGalSection(title, items) {
  if (!items.length) return '';
  const single = items.length === 1;
  const imgs = items.map(item =>
    `<div class="gal-img-wrap" onclick="openLightboxFromGlobal(${item.idx})">
      <img src="${item.src}" alt="" loading="lazy"
           onerror="this.src='https://placehold.co/300x200/e8f4f8/aaa?text=Photo'">
    </div>`
  ).join('');
  return `<div class="gal-section-title">${title} <span class="gal-section-count">${items.length} รูป</span></div>
<div class="gal-divider"></div>
<div class="gal-grid${single?' single':''}" style="margin-bottom:28px;">${imgs}</div>`;
}

};

<script>
function toggleFloat(){
  const btn=document.getElementById('floatBtn'),backdrop=document.getElementById('floatBackdrop'),mainBtn=document.getElementById('floatMainBtn');
  const isOpen=btn.classList.contains('expanded');
  if(isOpen){closeFloat();}else{btn.classList.add('expanded');mainBtn.classList.add('open');backdrop.classList.add('show');}
}
function closeFloat(){
  document.getElementById('floatBtn').classList.remove('expanded');
  document.getElementById('floatMainBtn').classList.remove('open');
  document.getElementById('floatBackdrop').classList.remove('show');
}
</script>
